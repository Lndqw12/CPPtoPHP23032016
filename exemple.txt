//Найти все трёхзначные числа, которые при увеличении на 1
//делятся на 2, при увеличении на 2 делятся на 3, при увеличении на 3
//делятся на 4, а при увеличении на 4 делятся на 5.(3 циклf for, цикл if
//для проверки.

// Для решения это задачи используем 
// в каждом из условий деление по модулю и сравние с нулем,
// если истинно все условия увеличим соответствующий счетчик
// 

#include "stdafx.h"
#include <iostream>   //кстати эти команды (включения) можно прописывать в "stdafx.h"
#include <math.h>     //


using namespace std;

int _tmain(int argc, _TCHAR* argv[])
{
	// нам понадобиться массив целых чисел
	int NumArray[900];  // массив от 100 до 999 трехзначные числа
		

	for (int count = 0; count < 900; count++)
		NumArray[count] = count + 100; // заполним массив числами от 100 до 999
	for (int i = 0; i <= 900; i++) //  for
	     {
			 if ((NumArray[i] + 1) % 2 == 0) // которые при увеличении на 1
				                             //делятся на 2,
			 {
				 if ((NumArray[i] + 2) % 3 == 0)  //при увеличении на 2 делятся на 3,
				 {
					 if ((NumArray[i] + 3) % 4 == 0) //при увеличении на 3
                                                     //делятся на 4, 
					    {
						 if ((NumArray[i] + 4) % 5 == 0) // а при увеличении на 4 делятся на 5.
							 cout << NumArray[i] << ","; // вывод результатов
						                                 // техзначные числа удовлетворяющих
						                                  // условию
				         }//if mod 4
				   } //if mod 3
			 } //if mod 2
	     } // for i

	system("pause");
	return 0;
}
